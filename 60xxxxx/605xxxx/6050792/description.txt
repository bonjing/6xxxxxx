-instructiuni clona
           spawning
	press "alt fire" (tab) and "look behind" (click mouse wheel)

	after spawned:
	
	lock/synchronize weapon
		press "alt fire" and "look behind" while on foot

	manually move clone
		press "alt fire" and "group contol forward" (g) while on foot

		while in manual mode:

		make clone drive car
			press "enter/exit" (f) while clone is near a car

			while clone is driving solo:

			make clone exit car
				press "group contol forward" while on foot

			get in passenger seat
				press "enter/exit" while next to the clone's car

		make clone attack person
			press "sprint" (space) while clone is near a person

		make clone attack all similar people
			press "fire" (left mouse button/lctrl) while clone is near a person

		don't attack similar people
			press "jump" (lshift/rcrtl) while clone is near a person

		put in clone's group (7 people max)
			press "sprint" when clone is near a member of the player's gang

		put in player's group
			press "sprint" when clone is near a member of the clone's gang
	
	switch seats with clone
		while you're driving and the clone's in the passenger seat, press "conversation yes" (y)

		while in passenger seat:

		change seats
			press "conversation yes"

		switch weapon
			press "alt fire" while not in driveby mode

		driveby -aim
			press "fire" or "handbrake" (space) or move the right analogue stick

		driveby -fire
			press "fire"

		set/remove destination
			place/remove marker on the map
		
		follow car
			press "conversation no" (n) while a car is in range

		change drive behavior
			press "conversation no" while no car is in range
		
		reset path (the clone follows nearest road)
			press (k)

		increase target speed
			press "accelerate" (w) in a car/bike
			
		decrease target speed
			press "decelerate" (s) in a car/bike

		increase target altitude
			press "accelerate" (w) in an airplane/helicopter
			
		decrease target altitude
			press "decelerate" (s) in an airplane/helicopter

		go into turret
			press "horn" (h)

			while in turret:

			exit turret
				press "crouch" (c)
		
		exit car
			press "enter/exit"

		manually correct clone's steering
			press"left/right" (a/d or left/right) to turn the car left or right
			press "steer up/steer down" (up/down) to make the car drive forward or backward

	Unfreeze clone
		press "group control forward" if the clone stops doing stuff

	destroy clone
		press (k)
